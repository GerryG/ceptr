497 int * _t_get_path(Tnode *t) {
498  Tnode *n;
499  // allocate an array to hold the
500  int s = sizeof(int)*10; // assume most trees are shallower than 10 nodes to prevent realloc
501  int *p = malloc(s);
502  int j,k,l,i=0,temp;
503 
504  // store the children's path index values into the array as we walk back up the tree to the root
505  for(n=t;n;) {
506  p[i] = _t_node_index(n);
507  n =_t_parent(n);
508  if (++i >= s) {
509  s*=2;p=realloc(p,s);} // realloc array if tree too deep
510  }
511  if (i > 2) {
512  // reverse the list by swapping elements going from the outside to the center
513  i-=2;
514  l = i+1;
515  k = i/2+1;
516  for(j=0;j<k;j++) {
517  temp = p[j];
518  p[j] = p[i];
519  p[i--] = temp;
520  }
521  }
522  else l = i-1;
523  p[l]= TREE_PATH_TERMINATOR;
524  return p;

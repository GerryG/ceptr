316 void _t_insert_at(T *t, int *path, T *i) {
317  T *c = _t_get(t,path);
318  int d = _t_path_depth(path)-1;
319  if (c) {
320  T *p = _t_parent(c);
321  if (!p) {
322  raise_error0("Can't insert into the root!");
323  }
324 
325  // first insert the new tree at the end just to use the code we
326  // already have for mallocing children, etc,
327  _t_add(p,i);
328 
329  // then shift the other children over
330  int j,l = _t_children(p);
331  j = l - path[d];
332  T **tp = &p->structure.children[l-1];
333  while(j--) {
334  *tp = *(tp-1);
335  tp--;
336  }
337  // and put the new tree where it belongs
338  *tp = i;
339  }
340  else {
341  // if path points to one beyond last child, we can simply add it.
342  if (path[d]>1) {
343  path[d]--;
344  c = _t_get(t,path);
345  if (c) {
346  T *p = _t_parent(c);
347  if (!p) {
348  raise_error0("Can't insert into the root!");
349  }
350  _t_add(p,i);
351  return;
352  }
353  }
354  else if (d==0) { // special case to insert into first child of empty root
355  _t_add(t,i);
356  return;
357  }

559 int * _t_get_path(T *t) {
560  T *n;
561  // allocate an array to hold the
562  int s = sizeof(int)*20; // assume most trees are shallower than 10 nodes to prevent realloc
563  int *p = malloc(s);
564  int j,k,l,i=0,temp;
565 
566  // store the children's path index values into the array as we walk back up the tree to the root
567  for(n=t;n;) {
568  p[i] = _t_node_index(n);
569  n =_t_parent(n);
570  if (++i >= s) {
571  s*=2;p=realloc(p,s);} // realloc array if tree too deep
572  }
573  if (i > 2) {
574  // reverse the list by swapping elements going from the outside to the center
575  i-=2;
576  l = i+1;
577  k = i/2+1;
578  for(j=0;j<k;j++) {
579  temp = p[j];
580  p[j] = p[i];
581  p[i--] = temp;
582  }
583  }
584  else l = i-1;

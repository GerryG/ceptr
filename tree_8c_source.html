334  T *c = _t_get(t,path);
335  int d = _t_path_depth(path)-1;
336  if (c) {
337  T *p = _t_parent(c);
338  if (!p) {
339  raise_error0("Can't insert into the root!");
340  }

342  // first insert the new tree at the end just to use the code we
343  // already have for mallocing children, etc,

596 int * _t_get_path(T *t) {
597  T *n;
598  // allocate an array to hold the
599  int s = sizeof(int)*20; // assume most trees are shallower than 10 nodes to prevent realloc
600  int *p = malloc(s);
601  int j,k,l,i=0,temp;
602 
603  // store the children's path index values into the array as we walk back up the tree to the root
604  for(n=t;n;) {
605  p[i] = _t_node_index(n);
606  n =_t_parent(n);
607  if (++i >= s) {
608  s*=2;p=realloc(p,s);} // realloc array if tree too deep
609  }
610  if (i > 2) {
611  // reverse the list by swapping elements going from the outside to the center
612  i-=2;
613  l = i+1;
614  k = i/2+1;
615  for(j=0;j<k;j++) {
616  temp = p[j];
617  p[j] = p[i];
618  p[i--] = temp;
619  }
620  }
621  else l = i-1;
622  p[l]= TREE_PATH_TERMINATOR;
623  return p;

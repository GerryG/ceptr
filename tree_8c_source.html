391  if (c) {

715 int * _t_get_path(T *t) {
716  T *n;
717  // allocate an array to hold the
718  int s = sizeof(int)*20; // assume most trees are shallower than 10 nodes to prevent realloc
719  int *p = malloc(s);
720  int j,k,l,i=0,temp;
721 
722  // store the children's path index values into the array as we walk back up the tree to the root
723  for(n=t;n;) {
724  p[i] = _t_node_index(n);
725  n =_t_parent(n);
726  if (++i >= s) {
727  s*=2;p=realloc(p,s);} // realloc array if tree too deep
728  }
729  if (i > 2) {
730  // reverse the list by swapping elements going from the outside to the center
731  i-=2;
732  l = i+1;
733  k = i/2+1;
734  for(j=0;j<k;j++) {
735  temp = p[j];
736  p[j] = p[i];
737  p[i--] = temp;
738  }
739  }

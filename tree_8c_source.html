415 void _t_insert_at(T *t, int *path, T *i) {
416  T *c = _t_get(t,path);
417  int d = _t_path_depth(path)-1;
418  if (c) {
419  T *p = _t_parent(c);
420  if (!p) {
421  raise_error("Can't insert into the root!");
422  }
423 
424  // first insert the new tree at the end just to use the code we
425  // already have for mallocing children, etc,

428  // then shift the other children over

616  T *t = parent;
617  bool done = false;
618  Structure st = NULL_STRUCTURE;
619  char *stn;
620  T *def,*p;
621  int state = bReadSymbol;
622  // Symbol expecting[100];
623  // T* expecting_def[100];
624  int level = 0;
625  while(!done) {
626  switch(state) {
627  case bReadSymbol:
628  param = va_arg(ap,Symbol);
629  if (semeq(param,NULL_SYMBOL)) {debug(D_TREE,"read NULL_SYMBOL\n");state=bPop;break;}
630  //expecting[level]
631  type = _getBuildType(sem,param,&st,&def);
632  //expectig_def[level] = def;
633  debug(D_TREE,"reading %s of type: %s with def:%s\n",_sem_get_name(sem,param),_sem_get_name(sem,type),_t2s(sem,def));
634  if (semeq(type,PROCESS_SIGNATURE)) {
635  state = bAddRoot;
636  }
637  else if (semeq(type,STRUCTURE_SYMBOL)) {
638  if (semeq(*(Symbol *)_t_surface(def),NULL_SYMBOL)) {
639  state = bReadBase;
640  }
641  else {
642  state = bAddRoot;
643  }
644  }
645  else if (semeq(type,NULL_STRUCTURE)) {
646  // null structure literal so just add it and pop
647  _t_newr(t,param);
648  state = bPop;
649  }
650  else if (semeq(type,STRUCTURE_OR)) {
651  state = bAddRoot;
652  }
653  else if (semeq(type,STRUCTURE_ZERO_OR_MORE) || semeq(type,STRUCTURE_ANYTHING) || semeq(type,STRUCTURE_SEQUENCE) || semeq(type,STRUCTURE_ONE_OR_MORE)) {
654  state = bAddRoot;
655  }
656  else {

682  j++;

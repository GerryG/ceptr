474  // allocate an array to hold the

476  int *p = malloc(s);

478 
479  // store the children's path index values into the array as we walk back up the tree to the root
480  for(n=t;n;) {

484  s*=2;p=realloc(p,s);} // realloc array if tree too deep
485  }

488  i-=2;

661  int *p = malloc(s);
662  int j,k,l,i=0,temp;
663 
664  // store the children's path index values into the array as we walk back up the tree to the root
665  for(n=t;n;) {

669  s*=2;p=realloc(p,s);} // realloc array if tree too deep

457  Tnode *n;
458  // allocate an array to hold the
459  int s = sizeof(int)*10; // assume most trees are shallower than 10 nodes to prevent realloc
460  int *p = malloc(s);
461  int j,k,l,i=0,temp;
462 
463  // store the children's path index values into the array as we walk back up the tree to the root
464  for(n=t;n;) {
465  p[i] = _t_node_index(n);
466  n =_t_parent(n);
467  if (++i >= s) {

469  }

472  i-=2;

474  k = i/2+1;

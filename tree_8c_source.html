428  T *c = _t_get(t,path);
429  int d = _t_path_depth(path)-1;
430  if (c) {
431  T *p = _t_parent(c);
432  if (!p) {
433  raise_error("Can't insert into the root!");
434  }
435 
436  // first insert the new tree at the end just to use the code we
437  // already have for mallocing children, etc,
438  _t_add(p,i);
439 
440  // then shift the other children over
441  int j,l = _t_children(p);
442  j = l - path[d];

444  while(j--) {
445  *tp = *(tp-1);
446  tp--;
447  }
448  // and put the new tree where it belongs
449  *tp = i;
450  }

628  T *t = parent;

630  Structure st = NULL_STRUCTURE;
631  char *stn;
632  T *def,*p;
633  int state = bReadSymbol;
634  // Symbol expecting[100];
635  // T* expecting_def[100];
636  int level = 0;
637  while(!done) {
638  switch(state) {
639  case bReadSymbol:
640  param = va_arg(ap,Symbol);
641  if (semeq(param,NULL_SYMBOL)) {debug(D_TREE,"read NULL_SYMBOL\n");state=bPop;break;}
642  //expecting[level]

644  //expectig_def[level] = def;

646  if (semeq(type,PROCESS_SIGNATURE)) {
647  state = bAddRoot;
648  }

652  }

655  }
656  }

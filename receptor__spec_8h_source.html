626 void testReceptorClock() {
627  Receptor *r = _r_makeClockReceptor(G_sem);
628  spec_is_str_equal(_td(r,r->root),"(RECEPTOR_INSTANCE (CONTEXT_NUM:4) (PARENT_CONTEXT_NUM:0) (RECEPTOR_STATE (FLUX (DEFAULT_ASPECT (EXPECTATIONS (EXPECTATION (CARRIER:CLOCK_TELL_TIME) (PATTERN (SEMTREX_SYMBOL_LITERAL (SEMTREX_SYMBOL:CLOCK_TELL_TIME))) (ACTION:respond with current time) (PARAMS) (END_CONDITIONS (UNLIMITED)))) (SIGNALS))) (PENDING_SIGNALS) (PENDING_RESPONSES) (RECEPTOR_ELAPSED_TIME:0)))");
629 
630  /*
631  The clock receptor should do two things: respond to CLOCK_TELL_TIME signals with the current time, and also allow you to plant a listener based on a semtrex for any kind of time you want. If you want the current time just plant a listener for TICK. If you want to listen for every second plant a listener on the Symbol literal SECOND, and the clock receptor will trigger the listener every time the SECOND changes. You can also listen for particular intervals and times by adding specificity to the semtrex, so to trigger a 3:30am action a-la-cron listen for: "/<TICK:(%HOUR=3,MINUTE=30)>"
632  @todo we should also make the clock receptor also respond to other semantic formats, i.e. so it's easy to listen for things like "on Wednesdays", or other semantic date/time identifiers.
633  */
634 
635  // send the clock receptor a "tell me the time" request
636  ReceptorAddress self = __r_get_self_address(r);
637 
638  T *req = _t_newr(0,REQUEST);
639  __r_make_addr(req,TO_ADDRESS,self);
640  _t_news(req,ASPECT_IDENT,DEFAULT_ASPECT);
641  _t_news(req,CARRIER,TICK);
642  _t_newr(req,CLOCK_TELL_TIME);
643  _t_news(req,RESPONSE_CARRIER,TICK);
644  T *run_tree = __p_build_run_tree(req,0);
645  _t_free(req);
646  _p_addrt2q(r->q,run_tree);
647 
648  // debug_enable(D_SIGNALS);
649 
650  while (r->q->contexts_count) {
651  _p_reduceq(r->q);
652 
653  // @todo fix this fake signal sending which only works here in this test because
654  // the signals are being sent to ourself!
655  T *signals = r->pending_signals;
656  while(_t_children(signals)>0) {
657  T *s = _t_detach_by_idx(signals,1);
658  Error err = _r_deliver(r,s);
659  if (err) raise_error("delivery error: %d",err);
660  }
661  }
662

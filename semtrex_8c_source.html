273  switch(s->type) {
274  case StateValue:
275  if (!t) return 0;
276  else {
277  size_t i;
278  if (s->data.value.length != _t_size(t));
279  p1 = s->data.value.value;
280  p2 = _t_surface(t);
281  for(i=s->data.value.length;i>0;i--) {
282  if (*p1++ != *p2++) return 0;
283  }
284  }
285  // no break to fall through and check symbol
286  case StateSymbol:
287  if (!t || (s->data.symbol != _t_symbol(t))) return 0;
288  // no break to fall through and handle transition and recursive calls
289  case StateAny:
290  if (!t) return 0;

293  case TransitionNextChild:
294  return __t_match(s->out,_t_next_sibling(t),r);

302  }
303  t = _t_next_sibling(t);
304  return __t_match(s->out,t,r);
305  break;
306  }
307  break;
308  case StateSplit:
309  if (__t_match(s->out,t,r)) return 1;
310  else return __t_match(s->out1,t,r);
311  break;
312  case StateGroup:
313  if (!r)
314  // if we aren't collecting up match results simply follow groups through
315  return __t_match(s->out,t,r);

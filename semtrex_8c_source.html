843  }
844  if (count > 1)
845  sprintf(b+strlen(b),"}");

849  case SEMTREX_SYMBOL_LITERAL_ID:

885  if (count > 1)
886  sprintf(b+strlen(b),"}");

903  sn = __dump_semtrex(defs,t,b);

911  sprintf(buf+strlen(buf),(_t_children(t) > has_child) ? "(%s)":"%s",sn);
912  break;
913  case SEMTREX_NOT_ID:
914  t = _t_child(s,1);
915  sn = __dump_semtrex(defs,t,b);
916  sprintf(buf,"~%s",sn);
917  break;
918  case SEMTREX_ZERO_OR_MORE_ID:

921  case SEMTREX_ONE_OR_MORE_ID:

929  // ignore "<unknown symbol"
930  if (*sn=='<')

934  break;
935  case SEMTREX_DESCEND_ID:

937  break;
938  case SEMTREX_WALK_ID:
939  sprintf(buf, "(%%%s)",__dump_semtrex(defs,_t_child(s,1),b));
940  break;
941  }
942  return buf;
943 }

115 T *_makeTestHTTPRequestTree() {
116  // manually build up a tree for the HTTP request:
117  // GET /groups/5/users.json?sort_by=last_name?page=2 HTTP/1.0
118  // Note that we put the version at the beginning of our tree just because in ceptr we
119  // do that as a best practice so that semtrex expectation matching can efficiently
120  // switch to different processing based on version numbers.
121  T *t = _t_new_root(HTTP_REQUEST);
122  T *t_version = _t_newr(t,HTTP_REQUEST_VERSION);
123  _t_newi(t_version,VERSION_MAJOR,1);
124  _t_newi(t_version,VERSION_MINOR,0);
125  T *t_method = _t_new(t,HTTP_REQUEST_METHOD,"GET",4);
126  T *t_path = _t_newr(t,HTTP_REQUEST_PATH);
127  T *t_segments = _t_newr(t_path,HTTP_REQUEST_PATH_SEGMENTS);
128  _t_new(t_segments,HTTP_REQUEST_PATH_SEGMENT,"groups",7);
129  _t_new(t_segments,HTTP_REQUEST_PATH_SEGMENT,"5",2);
130  T *t_file = _t_newr(t_path,HTTP_REQUEST_PATH_FILE);
131  _t_new(t_file,FILE_NAME,"users",6);
132  _t_new(t_file,FILE_EXTENSION,"json",5);
133  T *t_query = _t_newr(t_path,HTTP_REQUEST_PATH_QUERY);
134  T *t_params = _t_newr(t_query,HTTP_REQUEST_PATH_QUERY_PARAMS);
135  T *t_param1 = _t_newr(t_params,HTTP_REQUEST_PATH_QUERY_PARAM);
136  _t_new(t_param1,PARAM_KEY,"sort_by",8);
137  _t_new(t_param1,PARAM_VALUE,"last_name",10);
138  T *t_param2 = _t_newr(t_params,HTTP_REQUEST_PATH_QUERY_PARAM);
139  _t_new(t_param2,PARAM_KEY,"page",5);
140  _t_new(t_param2,PARAM_VALUE,"2",2);
141 
142  // confirm that we built the request right!
143  T *stx = _d_build_def_semtrex(test_HTTP_defs,HTTP_REQUEST,0);
144 
145  if (!_t_match(stx,t)) {raise_error0("BAD HTTP_REQUEST semtrex");}
146  _t_free(stx);
147 
148  return t;
149 }

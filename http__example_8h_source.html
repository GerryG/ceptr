65 Tnode *_makeTestHTTPRequestTree() {
67  // manually build up a tree for the HTTP request:
68  // GET /groups/5/users.json?sort_by=last_name?page=2 HTTP/1.0
69  // Note that we put the version at the beginning of our tree just because in ceptr we
70  // do that as a best practice so that semtrex expectation matching can efficiently
71  // switch to different processing based on version numbers.
72  Tnode *t = _t_new_root(TSYM_HTTP_REQUEST);
73  Tnode *t_version = _t_newr(t,TSYM_HTTP_REQUEST_VERSION);
74  _t_newi(t_version,TSYM_HTTP_REQUEST_VERSION_MAJOR,1);
75  _t_newi(t_version,TSYM_HTTP_REQUEST_VERSION_MINOR,0);
76  Tnode *t_method = _t_newi(t,TSYM_HTTP_REQUEST_METHOD,TEST_HTTP_METHOD_GET_VALUE);
77  Tnode *t_path = _t_newr(t,TSYM_HTTP_REQUEST_PATH);
78  Tnode *t_segments = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_SEGMENTS);
79  _t_new(t_segments,TSYM_HTTP_REQUEST_PATH_SEGMENT,"groups",7);
80  _t_new(t_segments,TSYM_HTTP_REQUEST_PATH_SEGMENT,"5",2);
81  Tnode *t_file = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_FILE);
82  _t_new(t_file,TSYM_HTTP_REQUEST_PATH_FILE_NAME,"users",6);
83  _t_new(t_file,TSYM_HTTP_REQUEST_PATH_FILE_EXTENSION,"json",5);
84  Tnode *t_query = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_QUERY);
85  Tnode *t_params = _t_newr(t_query,TSYM_HTTP_REQUEST_PATH_QUERY_PARAMS);
86  Tnode *t_param1 = _t_newr(t_params,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM);
87  _t_new(t_param1,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_KEY,"sort_by",8);
88  _t_new(t_param1,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_VALUE,"last_name",10);
89  Tnode *t_param2 = _t_newr(t_params,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM);
90  _t_new(t_param2,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_KEY,"page",5);
91  _t_new(t_param2,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_VALUE,"2",2);
92  return t;
93 }

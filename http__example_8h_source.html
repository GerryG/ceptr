76  // manually build up a tree for the HTTP request:
77  // GET /groups/5/users.json?sort_by=last_name?page=2 HTTP/1.0
78  // Note that we put the version at the beginning of our tree just because in ceptr we
79  // do that as a best practice so that semtrex expectation matching can efficiently
80  // switch to different processing based on version numbers.
81  Tnode *t = _t_new_root(TSYM_HTTP_REQUEST);
82  Tnode *t_version = _t_newr(t,TSYM_HTTP_REQUEST_VERSION);

84  _t_newi(t_version,TSYM_HTTP_REQUEST_VERSION_MINOR,0);

118 Tnode *_makeTestHTTPRequestTree() {
120  // manually build up a tree for the HTTP request:
121  // GET /groups/5/users.json?sort_by=last_name?page=2 HTTP/1.0
122  // Note that we put the version at the beginning of our tree just because in ceptr we
123  // do that as a best practice so that semtrex expectation matching can efficiently
124  // switch to different processing based on version numbers.
125  Tnode *t = _t_new_root(TSYM_HTTP_REQUEST);
126  Tnode *t_version = _t_newr(t,TSYM_HTTP_REQUEST_VERSION);
127  _t_newi(t_version,TSYM_HTTP_REQUEST_VERSION_MAJOR,1);
128  _t_newi(t_version,TSYM_HTTP_REQUEST_VERSION_MINOR,0);
129  Tnode *t_method = _t_newi(t,TSYM_HTTP_REQUEST_METHOD,TEST_HTTP_METHOD_GET_VALUE);
130  Tnode *t_path = _t_newr(t,TSYM_HTTP_REQUEST_PATH);
131  Tnode *t_segments = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_SEGMENTS);
132  _t_new(t_segments,TSYM_HTTP_REQUEST_PATH_SEGMENT,"groups",7);
133  _t_new(t_segments,TSYM_HTTP_REQUEST_PATH_SEGMENT,"5",2);
134  Tnode *t_file = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_FILE);
135  _t_new(t_file,TSYM_HTTP_REQUEST_PATH_FILE_NAME,"users",6);
136  _t_new(t_file,TSYM_HTTP_REQUEST_PATH_FILE_EXTENSION,"json",5);
137  Tnode *t_query = _t_newr(t_path,TSYM_HTTP_REQUEST_PATH_QUERY);
138  Tnode *t_params = _t_newr(t_query,TSYM_HTTP_REQUEST_PATH_QUERY_PARAMS);
139  Tnode *t_param1 = _t_newr(t_params,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM);
140  _t_new(t_param1,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_KEY,"sort_by",8);
141  _t_new(t_param1,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_VALUE,"last_name",10);
142  Tnode *t_param2 = _t_newr(t_params,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM);
143  _t_new(t_param2,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_KEY,"page",5);
144  _t_new(t_param2,TSYM_HTTP_REQUEST_PATH_QUERY_PARAM_VALUE,"2",2);
145  return t;
146 }

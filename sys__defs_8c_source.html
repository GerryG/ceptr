115  for(i=0;i<num_params;i++) {
116  T * t = va_arg(params,T *);
117  // if it's a SYMBOL_SET we need to just use the symbol type and
118  // can throw away the node
119  if (semeq(sym,STRUCTURE_SYMBOL_SET)) {
120  Symbol ss = *(Symbol *)_t_surface(t);
121  if (!is_symbol(ss)) raise_error("only structures allowed in SYMBOL_SET, def of %s",G_label);
122  _t_news(set,STRUCTURE_SYMBOL,ss);
123  _t_free(t);

125  // otherwise they should all be T *
126  else {
127  if (semeq(_t_symbol(t),STRUCTURE_SYMBOL)) {
128  Symbol ss = *(Symbol *)_t_surface(t);

131  if (!st) {

296  T *p = _t_child(r->defs.protocols,idx);

522 void __r_check_listener(T* processes,T *listener,T *signal,Q *q) {
523  T *signal_contents = (T *)_t_surface(_t_child(signal,2));
524 
525  T *e,*m;
526  e = _t_child(listener,1);
527  // if we get a match, create a run tree from the action, using the match and signal as the parameters
528  T *stx = _t_news(0,SEMTREX_GROUP,NULL_SYMBOL);
529  _t_add(stx,_t_clone(_t_child(e,1)));
530  if (_t_matchr(stx,signal_contents,&m)) {
531  T *rt=0;
532  T *action = _t_child(listener,2);
533  if (!action) {
534  raise_error0("null action in listener!");
535  }
536  if (semeq(_t_symbol(action),ACTION)) {
537  rt = _p_make_run_tree(processes,action,2,m,signal_contents);
538  _t_add(signal,rt);
539  _p_addrt2q(q,rt);
540  }
541  else {
542  // unblock the context
543  R *context = *(R**) _t_surface(action);
544  _p_unblock(q,context,m,signal_contents);
545  }
546  _t_free(m);
547  }
548  _t_free(stx);

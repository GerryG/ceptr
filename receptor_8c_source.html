530 char * __dump_semtrex(Tnode *s,char *buf) {
531  Symbol sym = _t_symbol(s);
532  char b[1000];
533  char *sn;
534  int i,c;
535  switch(sym) {
536  case SEMTREX_VALUE_LITERAL:
537  case SEMTREX_SYMBOL_LITERAL:
538  c = *(int *)_t_surface(s);
539  sn = _d_get_symbol_name(0,c);
540  // ignore "<unknown symbol"
541  if (*sn=='<')
542  sprintf(b,"%d",c);
543  else
544  sprintf(b,"%s",sn);
545  if (sym == SEMTREX_VALUE_LITERAL) {
546  sprintf(b+strlen(b),"=???");
547  }
548  __stxd_descend(s,b,buf);
549  break;
550  case SEMTREX_SYMBOL_ANY:
551  sprintf(b,".");
552  __stxd_descend(s,b,buf);
553  break;
554  case SEMTREX_SEQUENCE:
555  c = _t_children(s);
556  sn = buf;
557  for(i=1;i<=c;i++) {
558  sprintf(sn,i<c ? "%s,":"%s",__dump_semtrex(_t_child(s,i),b));
559  sn += strlen(sn);
560  }
561  break;
562  case SEMTREX_OR:
563  sprintf(buf,"(%s)|(%s)",__dump_semtrex(_t_child(s,1),b),__dump_semtrex(_t_child(s,2),b));
564  break;
565  case SEMTREX_ZERO_OR_MORE:
566  __stxd_multi("*",s,buf);
567  break;
568  case SEMTREX_ONE_OR_MORE:
569  __stxd_multi("+",s,buf);
570  break;
571  case SEMTREX_ZERO_OR_ONE:
572  __stxd_multi("?",s,buf);
573  break;
574  case SEMTREX_GROUP:
575  sn = _d_get_symbol_name(0,*(int *)_t_surface(s));
576  // ignore "<unknown symbol"
577  if (*sn=='<')
578  sprintf(buf, "{%s}",__dump_semtrex(_t_child(s,1),b));
579  else
580  sprintf(buf, "{%s: %s}",sn,__dump_semtrex(_t_child(s,1),b));
581  break;
582  }
583  return buf;
584 }

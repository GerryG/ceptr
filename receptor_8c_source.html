286 void _r_express_protocol(Receptor *r,int idx,Symbol sequence,Aspect aspect,T* handler) {
287  T *p = _t_child(r->defs.protocols,idx);
288  T *sequences = _t_child(p,2);
289  int j,c = _t_children(sequences);
290  for(j=1;j<=c;j++) {
291  T *seq = _t_child(sequences,j);
292  if (semeq(sequence,_t_symbol(seq))) {
293  // found the sequence, now get the first step
294  Symbol step1 = *(Symbol *)_t_surface(_t_child(seq,1));
295  // and search for the step definition in the list of steps
296  T *steps = _t_child(p,1);
297  c = _t_children(steps);
298  for(j=1;j<=c;j++) {
299  T *step = _t_child(steps,j);
300  if (semeq(_t_symbol(step),step1)) {
301  T *expect = _t_clone(_t_child(step,1));
302  T *params = _t_clone(_t_child(step,2));
303  T *act = _t_child(step,3);
304  // if there is no action, then assume its the sequence endpoint
305  // and use the handler in its place
307  if (act)
308  act = _t_clone(act);
309  else
310  act = handler;
311  //@todo turns out we don't use the carrier for anything yet, so
312  // we can just set it to a NULL_SYMBOL. This will have to change
313  // once we actually get carriers figured out
314  _r_add_listener(r,aspect,NULL_SYMBOL,expect,params,act);
315  return;
316  }
317  }
318  raise_error("step not found:%s",_r_get_symbol_name(r,step1));
319  }
320  }
321  raise_error("sequence not found:%s",_r_get_symbol_name(r,sequence));

439  T *signal_contents = (T *)_t_surface(_t_child(signal,2));
440 
441  T *e,*m;
442  e = _t_child(listener,1);
443  // if we get a match, create a run tree from the action, using the match and signal as the parameters
444  T *stx = _t_news(0,SEMTREX_GROUP,NULL_SYMBOL);
445  _t_add(stx,_t_clone(_t_child(e,1)));
446  if (_t_matchr(stx,signal_contents,&m)) {
447  T *rt=0;

449  if (!action) {

451  }

455  // this is actually the blocked phase of EXPECT_ACT. This could be a
456  // problem if we ever wanted our action to be an EXPECT_ACT process

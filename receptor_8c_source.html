584 void evaluateEndCondition(T *ec,bool *cleanup,bool *allow) {
585  *cleanup = false;
586  *allow = false;
587  int k = _t_children(ec);
588  while (k) {
589  T *c = _t_child(ec,k);
590  Symbol sym = _t_symbol(c);
591  if (semeq(sym,COUNT)) {
592  //@todo mutex!!
593  int *cP = (int *)_t_surface(c);
594  if (*cP <= 1) *cleanup = true;
595  if (*cP >= 1) *allow = true;
596  (*cP)--;
597  debug(D_SIGNALS,"decreasing count to: %d\n",*cP);
598  break; // this is final, even if there's a timeout
599  }
600  else if (semeq(sym,TIMEOUT_AT)) {
601  T *td = _t_child(c,1);
602  T *nw = _t_child(c,2);
603  int year = *(int *)_t_surface(_t_child(td,1))-1900;
604  int mon = *(int *)_t_surface(_t_child(td,2))-1;
605  int mday = *(int *)_t_surface(_t_child(td,3));
606  int hour = *(int *)_t_surface(_t_child(nw,1));
607  int min = *(int *)_t_surface(_t_child(nw,2));
608  int sec = *(int *)_t_surface(_t_child(nw,3));
609 
610  //debug(D_SIGNALS,"T: y:%d,m:%d,d:%d,h:%d,m:%d,s:%d\n")
611  time_t now_t;
612  time(&now_t);
613  struct tm now;
614  gmtime_r(&now_t, &now);
615  if ((year > now.tm_year) ||
616  (mon > now.tm_mon) ||
617  (mday > now.tm_mday) ||
618  (hour > now.tm_hour) ||
619  (min > now.tm_min) ||
620  (sec > now.tm_sec)) {
621  *allow = true;
622  }
623 
624  *cleanup = !*allow;
625  }
626  else if (semeq(sym,UNLIMITED)) {
627  *allow = true;
628  }
629  else {
630  raise_error("unknown end condition %s",t2s(c));
631  }
632 
633  k--;
634  }
635  debug(D_SIGNALS,"after end condition %s cleanup=%s allow=%s\n",t2s(ec),*cleanup?"true":"false",*allow?"true":"false");
636 }

645

651  debug(D_SIGNALS,"against expectation carrier %s\n",_td(q->r,e_carrier));
652

658  // if we get a match, create a run tree from the action, using the match and signal as the parameters

660  _t_add(stx,_t_clone(_t_child(pattern,1)));

662  debug(D_SIGNALS,"against %s\n",_td(q->r,stx));

667

704  }
705  else {
706  Process p = *(Process*) _t_surface(action);
707  T *params = _t_rclone(_t_child(expectation,ExpectationParamsIdx)); // __p_make_run_tree assumes rT nodes
708  _p_interpolate_from_match(params,m,signal_contents);

710  _t_free(params);

712  _p_addrt2q(q,rt);
713  }
714  _t_free(m);
715  }
716  if (cleanup) {
717  debug(D_SIGNALS,"cleaning up %s\n",_td(q->r,expectation));
718  _r_remove_expectation(q->r,expectation);
719  }

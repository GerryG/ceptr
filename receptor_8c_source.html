495  Tnode *m,*e,*l,*rt=0;
496 
497  Tnode *as = __r_get_signals(r,aspect);
498  Tnode *s = _t_newt(as,SIGNAL,signal_contents);
499 
500  // walk through all the listeners on the aspect and see if any expectations match this incoming signal
501  Tnode *ls = __r_get_listeners(r,aspect);
502  int i,c = _t_children(ls);
503  for(i=1;i<=c;i++) {
504  l = _t_child(ls,i);
505  e = _t_child(l,1);
506  // if we get a match, create a run tree from the action, using the match and signal as the parameters
507  if (_t_matchr(_t_child(e,1),signal_contents,&m)) {
508  Tnode *action = _t_child(l,2);

510  _t_free(m);

512  // for now just reduce the tree in place
514  int e = _p_reduce(r->defs,rt);
516  if (e) {
517  raise_error("got reduction error: %d",e);
518  }
519  }
520  }
521 
523  if (rt == 0) return 0;
524  else return s;

579 char * __dump_semtrex(Tnode *s,char *buf) {
580  Symbol sym = _t_symbol(s);
581  char b[1000];
582  char *sn;
583  int i,c;
584  switch(sym) {
585  case SEMTREX_VALUE_LITERAL:
586  case SEMTREX_SYMBOL_LITERAL:
587  c = *(int *)_t_surface(s);
588  sn = _d_get_symbol_name(0,c);
589  // ignore "<unknown symbol"
590  if (*sn=='<')
591  sprintf(b,"%d",c);
592  else
593  sprintf(b,"%s",sn);
594  if (sym == SEMTREX_VALUE_LITERAL) {
595  sprintf(b+strlen(b),"=???");
596  }
597  __stxd_descend(s,b,buf);
598  break;
599  case SEMTREX_SYMBOL_ANY:
600  sprintf(b,".");
601  __stxd_descend(s,b,buf);
602  break;
603  case SEMTREX_SEQUENCE:
604  c = _t_children(s);
605  sn = buf;
606  for(i=1;i<=c;i++) {
607  sprintf(sn,i<c ? "%s,":"%s",__dump_semtrex(_t_child(s,i),b));
608  sn += strlen(sn);
609  }
610  break;
611  case SEMTREX_OR:
612  sprintf(buf,"(%s)|(%s)",__dump_semtrex(_t_child(s,1),b),__dump_semtrex(_t_child(s,2),b));
613  break;
614  case SEMTREX_ZERO_OR_MORE:
615  __stxd_multi("*",s,buf);
616  break;
617  case SEMTREX_ONE_OR_MORE:
618  __stxd_multi("+",s,buf);
619  break;
620  case SEMTREX_ZERO_OR_ONE:
621  __stxd_multi("?",s,buf);
622  break;
623  case SEMTREX_GROUP:
624  sn = _d_get_symbol_name(0,*(int *)_t_surface(s));
625  // ignore "<unknown symbol"
626  if (*sn=='<')
627  sprintf(buf, "{%s}",__dump_semtrex(_t_child(s,1),b));
628  else
629  sprintf(buf, "{%s: %s}",sn,__dump_semtrex(_t_child(s,1),b));
630  break;
631  }
632  return buf;
633 }

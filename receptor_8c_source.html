557 T * _r_deliver(Receptor *r, T *signal) {
558  T *m,*e,*l,*rt=0;
559 
560  T *signal_contents = (T *)_t_surface(_t_child(signal,2));
561  T *envelope = _t_child(signal,1);
562  Aspect aspect = *(Aspect *)_t_surface(_t_child(envelope,3));
563 
564  T *as = __r_get_signals(r,aspect);
565 
566  _t_add(as,signal);
567 
568  // walk through all the listeners on the aspect and see if any expectations match this incoming signal
569  T *ls = __r_get_listeners(r,aspect);
570  int i,c = _t_children(ls);
571  for(i=1;i<=c;i++) {
572  l = _t_child(ls,i);
573  e = _t_child(l,1);
574  // if we get a match, create a run tree from the action, using the match and signal as the parameters
575  if (_t_matchr(_t_child(e,1),signal_contents,&m)) {
576  T *action = _t_child(l,2);
577  rt = _p_make_run_tree(r->defs.processes,action,2,m,signal_contents);
578  _t_free(m);
579  _t_add(signal,rt);
580  // for now just reduce the tree in place
582  int e = _p_reduce(r->defs,rt);
584  if (e) {
585  raise_error("got reduction error: %d",e);
586  }
587  }
588  }
589 
591  if (rt == 0) return 0;
592  else return _t_child(_t_child(signal,3),1);

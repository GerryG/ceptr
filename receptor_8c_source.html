488 Tnode * _r_send(Receptor *r,Receptor *from,Aspect aspect, Tnode *signal_contents) {
489  Tnode *m,*e,*l,*rt=0;
490 
491  Tnode *as = __r_get_signals(r,aspect);
492  Tnode *s = _t_newt(as,SIGNAL,signal_contents);
493 
494  // walk through all the listeners on the aspect and see if any expectations match this incoming signal
495  Tnode *ls = __r_get_listeners(r,aspect);
496  int i,c = _t_children(ls);
497  for(i=1;i<=c;i++) {
498  l = _t_child(ls,i);
499  e = _t_child(l,1);
500  // if we get a match, create a run tree from the action, using the match and signal as the parameters
501  if (_t_matchr(_t_child(e,1),signal_contents,&m)) {
502  Tnode *action = _t_child(l,2);
503  rt = _p_make_run_tree(r->defs.processes,action,2,m,signal_contents);
504  _t_free(m);
505  _t_add(s,rt);
506  // for now just reduce the tree in place
508  int e = _p_reduce(r->defs,rt);
510  if (e) {
511  raise_error("got reduction error: %d",e);
512  }
513  }
514  }
515 
517  if (rt == 0) return 0;
518  else return s;

573 char * __dump_semtrex(Tnode *s,char *buf) {
574  Symbol sym = _t_symbol(s);
575  char b[1000];
576  char *sn;
577  int i,c;
578  switch(sym) {
579  case SEMTREX_VALUE_LITERAL:
580  case SEMTREX_SYMBOL_LITERAL:
581  c = *(int *)_t_surface(s);
582  sn = _d_get_symbol_name(0,c);
583  // ignore "<unknown symbol"
584  if (*sn=='<')
585  sprintf(b,"%d",c);
586  else
587  sprintf(b,"%s",sn);
588  if (sym == SEMTREX_VALUE_LITERAL) {
589  sprintf(b+strlen(b),"=???");
590  }
591  __stxd_descend(s,b,buf);
592  break;
593  case SEMTREX_SYMBOL_ANY:
594  sprintf(b,".");
595  __stxd_descend(s,b,buf);
596  break;
597  case SEMTREX_SEQUENCE:
598  c = _t_children(s);
599  sn = buf;
600  for(i=1;i<=c;i++) {
601  sprintf(sn,i<c ? "%s,":"%s",__dump_semtrex(_t_child(s,i),b));
602  sn += strlen(sn);
603  }
604  break;
605  case SEMTREX_OR:
606  sprintf(buf,"(%s)|(%s)",__dump_semtrex(_t_child(s,1),b),__dump_semtrex(_t_child(s,2),b));
607  break;
608  case SEMTREX_ZERO_OR_MORE:
609  __stxd_multi("*",s,buf);
610  break;
611  case SEMTREX_ONE_OR_MORE:
612  __stxd_multi("+",s,buf);
613  break;
614  case SEMTREX_ZERO_OR_ONE:
615  __stxd_multi("?",s,buf);
616  break;
617  case SEMTREX_GROUP:
618  sn = _d_get_symbol_name(0,*(int *)_t_surface(s));
619  // ignore "<unknown symbol"
620  if (*sn=='<')
621  sprintf(buf, "{%s}",__dump_semtrex(_t_child(s,1),b));
622  else
623  sprintf(buf, "{%s: %s}",sn,__dump_semtrex(_t_child(s,1),b));
624  break;
625  }
626  return buf;
627 }

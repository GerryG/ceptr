243 void __d_tsig(SemTable *sem,T *code, T *tsig,TreeHash *hashes) {
244  if (semeq(_t_symbol(code),SLOT)) {
245  TreeHash h;
246  T *c = NULL;
247  // slot children aren't part of the signature they are part of the
248  // the code that needs to be searched!
249  if (c = _t_find(code,SLOT_CHILDREN)) {
250  int i = _t_node_index(c);
251  c = _t_clone(code);
252  T *sc = _t_detach_by_idx(c,i);
253  __d_tsig(sem,sc,tsig,hashes);
254  _t_free(sc);
255  _t_hash(sem,c);
256  }
257  else h = _t_hash(sem,code);
258  // check for duplicates
259  int i=0;
260  while(hashes[i] && hashes[i]!=h) {
261  i++;
262  if (i == MAX_HASHES) raise_error("whoa! too many slots in code template");
263  }
264  // if its unique then add it.
265  if (!hashes[i]) {
266  hashes[i] = h;
267  if (!c) c = _t_clone(code); // clone it if it wasn't cloned above
268  c->contents.symbol = EXPECTED_SLOT;
269  _t_add(tsig,c);
270  c = NULL;
271  }
272  if (c) _t_free(c);
273  }
274  else {
275  DO_KIDS(code,__d_tsig(sem,_t_child(code,i),tsig,hashes));
276  }
277 }

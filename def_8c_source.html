380  break;

390  sprintf(buf," (<unknown:%d>",s);

392  sprintf(buf," (%s",n);
393  }
394  }
395  }
396  for(i=1;i<=_t_children(t);i++) __t_dump(defs,_t_child(t,i),level+1,buf+strlen(buf));

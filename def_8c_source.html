324 char * __t_dump(Defs *defs,T *t,int level,char *buf) {
325  T *structures = defs ? defs->structures : 0;
326  T *symbols = defs ? defs->symbols : 0;
327  T *processes = defs ? defs->processes : 0;
328  if (!t) return "";
329  Symbol s = _t_symbol(t);
330  char b[255];
331  char tbuf[2000];
332  int i;
333  char *c;
334  Xaddr x;
335  if (is_process(s)) {
336  sprintf(buf," (process:%s",_d_get_process_name(processes,s));
337  }
338  else {
339  char *n = _d_get_symbol_name(symbols,s);
340  Structure st = _d_get_symbol_structure(symbols,s);
341  if (!is_sys_structure(st)) {
342  // if it's not a system structure, it's composed, so all we need to do is
343  // print out the symbol name, and the reset will take care of itself
344  sprintf(buf," (%s",n);
345  }
346  else {
347  switch(st.id) {
348  case CSTRING_ID:
349  sprintf(buf," (%s:%s",n,(char *)_t_surface(t));
350  break;
351  case CHAR_ID:
352  sprintf(buf," (%s:'%c'",n,*(char *)_t_surface(t));
353  break;
354  case BOOLEAN_ID:
355  case INTEGER_ID:
356  sprintf(buf," (%s:%d",n,*(int *)_t_surface(t));
357  break;
358  case SYMBOL_ID:
359  c = _d_get_symbol_name(symbols,*(Symbol *)_t_surface(t));
360  sprintf(buf," (%s:%s",n,c?c:"<unknown>");
361  break;
362  case STRUCTURE_ID:
363  c = _d_get_structure_name(structures,*(Structure *)_t_surface(t));
364  sprintf(buf," (%s:%s",n,c?c:"<unknown>");
365  break;
366  case PROCESS_ID:
367  c = _d_get_process_name(processes,*(Process *)_t_surface(t));
368  sprintf(buf," (%s:%s",n,c?c:"<unknown>");
369  break;
370  case TREE_PATH_ID:
371  sprintf(buf," (%s:%s",n,_t_sprint_path((int *)_t_surface(t),b));
372  break;
373  case XADDR_ID:
374  x = *(Xaddr *)_t_surface(t);
375  sprintf(buf," (%s:%s.%d",n,_d_get_symbol_name(symbols,x.symbol),x.addr);
376  break;
377  case TREE_ID:
378  if (t->context.flags & TFLAG_SURFACE_IS_TREE) {
379  c = __t_dump(defs,(T *)_t_surface(t),0,tbuf);
380  sprintf(buf," (%s:{%s}",n,c);
381  break;
382  }
383  case RECEPTOR_ID:
384  if (t->context.flags & (TFLAG_SURFACE_IS_TREE+TFLAG_SURFACE_IS_RECEPTOR)) {
385  c = __t_dump(defs,((Receptor *)_t_surface(t))->root,0,tbuf);
386  sprintf(buf," (%s:{%s}",n,c);
387  break;
388  }
389  case SCAPE_ID:
390  if (t->context.flags & TFLAG_SURFACE_IS_SCAPE) {
391  Scape *sc = (Scape *)_t_surface(t);
392  sprintf(buf," (%s:key %s,data %s",n,_d_get_symbol_name(symbols,sc->key_source),_d_get_symbol_name(symbols,sc->data_source));
393  break;
394  }
395  default:
396  if (semeq(s,SEMTREX_MATCH_CURSOR)) {
397  c = __t_dump(defs,*(T **)_t_surface(t),0,tbuf);
398  //c = "null";
399  sprintf(buf," (%s:{%s}",n,c);
400  break;
401  }
402  if (n == 0)
403  sprintf(buf," (<unknown:%d.%d.%d>",s.context,s.flags,s.id);
404  else
405  sprintf(buf," (%s",n);
406  }
407  }
408  }
409  DO_KIDS(t,__t_dump(defs,_t_child(t,i),level+1,buf+strlen(buf)));
410  sprintf(buf+strlen(buf),")");
411  return buf;
412 }

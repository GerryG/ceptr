284 char * __t_dump(Tnode *symbols,Tnode *t,int level,char *buf) {
285  if (!t) return "";
286  Symbol s = _t_symbol(t);
287  char b[255];
288  char tbuf[2000];
289  int i;
290  char *n = _d_get_symbol_name(symbols,s);
291  char *c;
292  Xaddr x;
293  if (is_process(s)) {
294  sprintf(buf," (process:%s",_d_get_process_name(0,s));
295  }
296  else {
297  Structure st = _d_get_symbol_structure(symbols,s);
298  switch(st) {
299  case CSTRING:
300  sprintf(buf," (%s:%s",n,(char *)_t_surface(t));
301  break;
302  case BOOLEAN:
303  case INTEGER:
304  sprintf(buf," (%s:%d",n,*(int *)_t_surface(t));
305  break;
306  case SYMBOL:
307  c = _d_get_symbol_name(symbols,*(int *)_t_surface(t));
308  sprintf(buf," (%s:%s",n,c?c:"<unknown>");
309  break;
310  case TREE_PATH:
311  sprintf(buf," (%s:%s",n,_t_sprint_path((int *)_t_surface(t),b));
312  break;
313  case XADDR:
314  x = *(Xaddr *)_t_surface(t);
315  sprintf(buf," (%s:%s.%d",n,_d_get_symbol_name(symbols,x.symbol),x.addr);
316  break;
317  case TREE:
318  if (t->context.flags == TFLAG_SURFACE_IS_TREE) {
319  c = __t_dump(symbols,(Tnode *)_t_surface(t),0,tbuf);
320  sprintf(buf," (%s:{%s}",n,c);
321  break;
322  }
323  case RECEPTOR:
324  if (t->context.flags == TFLAG_SURFACE_IS_TREE+TFLAG_SURFACE_IS_RECEPTOR) {
325  c = __t_dump(symbols,((Receptor *)_t_surface(t))->root,0,tbuf);
326  sprintf(buf," (%s:{%s}",n,c);
327  break;
328  }
329  default:
330  switch(s) {
331  case LISTENER:
332  c = _d_get_symbol_name(symbols,*(int *)_t_surface(t));
333  sprintf(buf," (%s on %s",n,c?c:"<unknown>");
334  break;
335  default:
336  if (n == 0)
337  sprintf(buf," (<unknown:%d>",s);
338  else
339  sprintf(buf," (%s",n);
340  }
341  }
342  }
343  for(i=1;i<=_t_children(t);i++) __t_dump(symbols,_t_child(t,i),level+1,buf+strlen(buf));
344  sprintf(buf+strlen(buf),")");
345  return buf;
346 }

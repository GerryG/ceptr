381 Error _p_step(Defs *defs, R **contextP) {
382  R *context = *contextP;
383 
384  switch(context->state) {
385  case noReductionErr:
386  case Block:
387  raise_error0("whoa!"); // shouldn't be calling step if Done or noErr or Block
388  break;
389  case Pop:
390 
391  // if this was the successful reduction by an error handler
392  // move the value to the 1st child
393  if (context->err) {
394  T *t = _t_detach_by_idx(context->run_tree,3);
395  if (t) {
396  _t_replace(context->run_tree,1,t);
397  context->err = noReductionErr;
398  }
399  }
400 
401  // if this is top caller on the stack then we are completely done
402  if (!context->caller) {
403  context->state = Done;
404  break;
405  }
406  else {
407  // otherwise pop the context
408  R *ctx = context;
409  context = context->caller; // set the new context
410 
411  if (!ctx->err) {
412  // get results of the run_tree
413  T *np = _t_detach_by_idx(ctx->run_tree,1);
414  _t_replace(context->parent,context->idx,np); // replace the process call node with the result
415  rt_cur_child(np) = RUN_TREE_EVALUATED;
416  context->node_pointer = np;
417  context->state = Eval; // or possible ascend??
418  }
419  else context->state = ctx->err;
420  // cleanup
421  _t_free(ctx->run_tree);
422  free(ctx);
423  context->callee = 0;
424  *contextP = context;
425  }
426 
427  break;
428  case Eval:
429  {
430  T *np = context->node_pointer;
431  if (!np) {
432  raise_error0("Whoa! Null node pointer");
433  }
434  Process s = _t_symbol(np);
435 
436  if (semeq(s,PARAM_REF)) {
437  T *param = _t_get(context->run_tree,(int *)_t_surface(np));
438  if (!param) {
439  raise_error0("request for non-existent param");
440  }
441  context->node_pointer = np = _t_rclone(param);
442  _t_replace(context->parent, context->idx,np);
443  s = _t_symbol(np);
444  }
445  // @todo what if the replaced parameter is itself a PARAM_REF tree ??
446 
447  // if this node is not a process, i.e. it's data, then we are done descending
448  // and it will be the result so ascend
449  if (!is_process(s)) {
450  context->state = Ascend;
451  }
452  else {
453  int c = _t_children(np);
454  if (c == rt_cur_child(np) || semeq(s,QUOTE)) {
455  // if the current child == the child count this means
456  // all the children have been processed, so we can evaluate this process
457  // if the process is QUOTE that's a special case and we evaluate it
458  // immediately without descending.
459  if (!is_sys_process(s)) {
460  // if it's user defined process then we check the signature and then make
461  // a new run-tree run that process
462  Error e = __p_check_signature(defs,s,np);
463  if (e) context->state = e;
464  else {
465  T *run_tree = __p_make_run_tree(defs->processes,s,np);
466  context->state = Pushed;
467  *contextP = __p_make_context(run_tree,context);
468  }
469  }
470  else {
471  // if it's a sys process we can just reduce it in and then ascend
472  // or move to the error handling state
473  Error e = __p_reduce_sys_proc(context,s,np);
474  context->state = e ? e : Ascend;
475  }
476  }
477  else if(c) {
478  //descend and increment the current child we're working on!
479  context->state = Descend;
480  }
481  else {
482  raise_error0("whoa! brain fart!");
483  }
484  }
485  }
486  break;
487  case Ascend:
488  rt_cur_child(context->node_pointer) = RUN_TREE_EVALUATED;
489  context->node_pointer = context->parent;
490  context->parent = _t_parent(context->node_pointer);
491  if (!context->parent || context->parent == context->run_tree) {
492  context->idx = 1;
493  }
494  else context->idx = rt_cur_child(context->parent);
495  if (context->node_pointer == context->run_tree)
496  context->state = Pop;
497  else
498  context->state = Eval;
499  break;
500  case Descend:
501  context->parent = context->node_pointer;
502  context->idx = ++rt_cur_child(context->node_pointer);
503  context->node_pointer = _t_child(context->node_pointer,context->idx);
504  context->state = Eval;
505  break;
506  default:
507  context->err = context->state;
508  if (_t_children(context->run_tree) <= 2) {
509  // no error handler so just return the error
510  context->state = Pop;
511  }
512  else {
513  // the first parameter to the error code is always a reduction error
514  // which gets added on as the 4th child of the run tree when the
515  // error happens.
516  T *ps = _t_newr(context->run_tree,PARAMS);
517 
518  //@todo: fix this so we don't actually use an error value that
519  // then has to be translated into a symbol, but rather so that we
520  // can programatically calculate the symbol.
521  Symbol se;
522  switch(context->state) {
523  case tooFewParamsReductionErr: se=TOO_FEW_PARAMS_ERR;break;
524  case tooManyParamsReductionErr: se=TOO_MANY_PARAMS_ERR;break;
525  case signatureMismatchReductionErr: se=SIGNATURE_MISMATCH_ERR;break;
526  case notProcessReductionError: se=NOT_A_PROCESS_ERR;break;
527  case notInSignalContextReductionError: se=NOT_IN_SIGNAL_CONTEXT_ERR;

531  se = *(Symbol *)_t_surface(_t_child(context->node_pointer,1));
532  break;
533  default: raise_error("unknown reduction error: %d",context->state);
534  }
535  T *err = __t_new(ps,se,0,0,sizeof(rT));
536  int *path = _t_get_path(context->node_pointer);
537  _t_new(err,ERROR_LOCATION,path,sizeof(int)*(_t_path_depth(path)+1));
538  free(path);
539 
540  // switch the node_pointer to the top of the error handling routine
541  context->node_pointer = _t_child(context->run_tree,3);
542  context->idx = 3;
543  context->parent = context->run_tree;
544 
545  context->state = Eval;
546  }
547  }
548  return context->state;
549 }

296 Error _p_step(Defs *defs, R **contextP) {
297  R *context = *contextP;
298 
299  switch(context->state) {
300  case noReductionErr:
301  raise_error0("whoa!"); // shouldn't be calling step if Done or noErr
302  break;
303  case Pop:
304 
305  // if this was the successful reduction by an error handler
306  // move the value to the 1st child
307  if (context->err) {
308  T *t = _t_detach_by_idx(context->run_tree,3);
309  if (t) {
310  _t_replace(context->run_tree,1,t);
311  context->err = noReductionErr;
312  }
313  }
314 
315  // if this is top caller on the stack then we are completely done
316  if (!context->caller) {
317  context->state = Done;
318  break;
319  }
320  else {
321  // otherwise pop the context
322  R *ctx = context;
323  context = context->caller; // set the new context
324 
325  if (!ctx->err) {
326  // get results of the run_tree
327  T *np = _t_detach_by_idx(ctx->run_tree,1);
328  _t_replace(context->parent,context->idx,np); // replace the process call node with the result
329  rt_cur_child(np) = RUN_TREE_EVALUATED;
330  context->node_pointer = np;
331  context->state = Eval; // or possible ascend??
332  }
333  else context->state = ctx->err;
334  // cleanup
335  _t_free(ctx->run_tree);
336  free(ctx);
337  context->callee = 0;
338  *contextP = context;
339  }
340 
341  break;
342  case Eval:
343  {
344  T *np = context->node_pointer;
345 
346  Process s = _t_symbol(np);
347 
348  if (semeq(s,PARAM_REF)) {
349  T *param = _t_get(context->run_tree,(int *)_t_surface(np));
350  if (!param) {
351  raise_error0("request for non-existent param");
352  }
353  context->node_pointer = np = _t_rclone(param);
354  _t_replace(context->parent, context->idx,np);
355  s = _t_symbol(np);
356  }
357  // @todo what if the replaced parameter is itself a PARAM_REF tree ??
358 
359  // if this node is not a process, i.e. it's data, then we are done descending
360  // and it will be the result so ascend
361  if (!is_process(s)) {
362  context->state = Ascend;
363  }
364  else {
365  int c = _t_children(np);
366  if (c == rt_cur_child(np)) {
367  // if the current child == the child count this means
368  // all the children have been processed, so we can evaluate this process
369  if (!is_sys_process(s)) {
370  // if it's user defined process then we check the signature and then make
371  // a new run-tree run that process
372  Error e = __p_check_signature(defs,s,np);
373  if (e) context->state = e;
374  else {
375  T *run_tree = __p_make_run_tree(defs->processes,s,np);
376  context->state = Pushed;
377  *contextP = __p_make_context(run_tree,context);
378  }
379  }
380  else {
381  // if it's a sys process we can just reduce it in and then ascend
382  // or move to the error handling state
383  Error e = __p_reduce_sys_proc(context,s,np);
384  context->state = e ? e : Ascend;
385  }
386  }
387  else if(c) {
388  //descend and increment the current child we're working on!
389  context->state = Descend;
390  }
391  else {
392  raise_error0("whoa! brain fart!");
393  }
394  }
395  }
396  break;
397  case Ascend:
398  rt_cur_child(context->node_pointer) = RUN_TREE_EVALUATED;
399  context->node_pointer = context->parent;
400  context->parent = _t_parent(context->node_pointer);
401  if (!context->parent || context->parent == context->run_tree) {
402  context->idx = 1;
403  }
404  else context->idx = rt_cur_child(context->parent);
405  if (context->node_pointer == context->run_tree)
406  context->state = Pop;
407  else
408  context->state = Eval;
409  break;
410  case Descend:
411  context->parent = context->node_pointer;
412  context->idx = ++rt_cur_child(context->node_pointer);
413  context->node_pointer = _t_child(context->node_pointer,context->idx);
414  context->state = Eval;
415  break;
416  default:
417  context->err = context->state;
418  if (_t_children(context->run_tree) <= 2) {
419  // no error handler so just return the error
420  context->state = Pop;
421  }
422  else {
423  // the first parameter to the error code is always a reduction error
424  // which gets added on as the 4th child of the run tree when the
425  // error happens.
426  T *ps = _t_newr(context->run_tree,PARAMS);
427

430  // can programatically calculate the symbol.

432  switch(context->state) {
433  case tooFewParamsReductionErr: se=TOO_FEW_PARAMS_ERR;break;
434  case tooManyParamsReductionErr: se=TOO_MANY_PARAMS_ERR;break;
435  case badSignatureReductionErr: se=BAD_SIGNATURE_ERR;break;
436  case notProcessReductionError: se=NOT_A_PROCESS_ERR;break;
437  case notInSignalContextReductionError: se=NOT_IN_SIGNAL_CONTEXT_ERR;

441  break;
442  default: raise_error("unknown reduction error: %d",context->state);
443  }
444  T *err = __t_new(ps,se,0,0,sizeof(rT));
445  int *path = _t_get_path(context->node_pointer);
446  _t_new(err,ERROR_LOCATION,path,sizeof(int)*(_t_path_depth(path)+1));
447  free(path);
448

450  context->node_pointer = _t_child(context->run_tree,3);
451  context->idx = 3;
452  context->parent = context->run_tree;
453 
454  context->state = Eval;
455  }
456  }
457  return context->state;
458 }

785 Error _p_step(Q *q, R **contextP) {
786  Defs *defs = &q->r->defs;
787  R *context = *contextP;
788 
789  switch(context->state) {
790  case noReductionErr:
791  case Block:
792  raise_error("whoa, virtual states can't be executed!"); // shouldn't be calling step if Done or noErr or Block
793  break;
794  case Pop:
795  // if this was the successful reduction by an error handler
796  // move the value to the 1st child
797  if (context->err) {
798  T *t = _t_detach_by_idx(context->run_tree,3);
799  if (t) {
800  _t_replace(context->run_tree,1,t);
801 
802  context->err = noReductionErr;
803  }
804  }
805 
806  // if this is top caller on the stack then we are completely done
807  if (!context->caller) {
808  context->state = Done;
809  break;
810  }
811  else {
812  // otherwise pop the context
813  R *ctx = context;
814  context = context->caller; // set the new context
815 
816  if (!ctx->err) {
817  // get results of the run_tree
818  T *np = _t_detach_by_idx(ctx->run_tree,1);
819  _t_replace(context->parent,context->idx,np); // replace the process call node with the result
820  set_rt_cur_child(q->r,np,RUN_TREE_EVALUATED);
821  context->node_pointer = np;
822  context->state = Eval; // or possible ascend??
823  }
824  else context->state = ctx->err;
825  // cleanup
826  _t_free(ctx->run_tree);
827  free(ctx);
828  context->callee = 0;
829  *contextP = context;
830  }
831 
832  break;
833  case Eval:
834  {
835  T *np = context->node_pointer;
836  if (!np) {
837  raise_error("Whoa! Null node pointer");
838  }
839  Process s = _t_symbol(np);
840 
841  if (semeq(s,PARAM_REF)) {
842  T *param = _t_get(context->run_tree,(int *)_t_surface(np));
843  if (!param) {
844  raise_error("request for non-existent param");
845  }
846  context->node_pointer = np = _t_rclone(param);
847  _t_replace(context->parent, context->idx,np);
848  s = _t_symbol(np);
849  }
851 
852  int count = _t_children(np);
853  if (!is_process(s)) {
854 
855  // if this node is not a process, i.e. it's data, then either we
856  // are done descending and the current items will be the result so ascend
857  // or if we are doing deep param_ref searching, then search the entire tree
858  // @todo increase efficiency by adding some instruction to allow the coder choose, see #39
859 #ifndef RUN_TREE_SHALLOW_PARAM_REF_SEARCH
860  if (count && (count != get_rt_cur_child(q->r,np)))
861  context->state = Descend;
862  else
863 #endif
864  context->state = Ascend;
865  }
866  else
867  {
868  if (semeq(s,ITERATE)) {
869  // if first time we are hitting this replication
870  // then we need to set it up
871  // raise(SIGINT);
872  if (_t_size(np) == 0) {
873  // sanity check
874  if (_t_children(np) != 3) {raise_error("ITERATE must have 3 params");}
875  // create a copy of the code and stick it in the replication state struct
876  ReplicationState *state = malloc(sizeof(ReplicationState));
877  state->phase = EvalCondition;
878  state->code = _t_rclone(np);
879  state->type = ReplicateTypeUnknown;
880  *((ReplicationState **)&np->contents.surface) = state;
881  np->contents.size = sizeof(ReplicationState *);
882  // we start in condition phase so throw away the code copy
883  T *x = _t_detach_by_idx(np,3);
884  _t_free(x);
885  }
886  }
887  if (count == get_rt_cur_child(q->r,np) || semeq(s,QUOTE)) {
888  // if the current child == the child count this means
889  // all the children have been processed, so we can evaluate this process
890  // if the process is QUOTE that's a special case and we evaluate it
891  // immediately without descending.
892  if (!is_sys_process(s)) {
893  // if it's user defined process then we check the signature and then make
894  // a new run-tree run that process
895  Error e = __p_check_signature(defs,s,np);
896  if (e) context->state = e;
897  else {
898  T *run_tree = __p_make_run_tree(defs->processes,s,np);
899  context->state = Pushed;
900  *contextP = __p_make_context(run_tree,context,context->id);
901  }
902  }
903  else {
904  // if it's a sys process we can just reduce it in and then ascend
905  // or move to the error handling state
906  Error e = __p_reduce_sys_proc(context,s,np,q);
907  context->state = e ? e : Ascend;
908  }
909  }
910  else if(count) {
911  //descend and increment the current child we're working on!
912  context->state = Descend;
913  }
914  else {
915  raise_error("whoa! brain fart! on %d,%s",count,_t2s(&q->r->defs,np));
916  }
917  }
918  }
919  break;
920  case Ascend:
921  set_rt_cur_child(q->r,context->node_pointer,RUN_TREE_EVALUATED);
922  context->node_pointer = context->parent;
923  context->parent = _t_parent(context->node_pointer);
924  if (!context->parent || context->parent == context->run_tree || (context->node_pointer == context->run_tree)) {
925  context->idx = 1;
926  }
927  else {
928  context->idx = get_rt_cur_child(q->r,context->parent);
929  }
930  if (context->node_pointer == context->run_tree)
931  context->state = Pop;
932  else
933  context->state = Eval;
934  break;
935  case Descend:
936  context->parent = context->node_pointer;
937  rt_check(q->r,context->node_pointer);
938  context->idx = ++rt_cur_child(context->node_pointer);
939  context->node_pointer = _t_child(context->node_pointer,context->idx);
940  context->state = Eval;
941  break;
942  default:
943  context->err = context->state;
944  if (_t_children(context->run_tree) <= 2) {
945  // no error handler so just return the error
946  context->state = Pop;
947  }
948  else {
949  // the first parameter to the error code is always a reduction error
950  // which gets added on as the 4th child of the run tree when the
951  // error happens.
952  T *ps = _t_newr(context->run_tree,PARAMS);
953  T *extra = NULL;
954  //@todo: fix this so we don't actually use an error value that
955  // then has to be translated into a symbol, but rather so that we
956  // can programatically calculate the symbol.
957  // or, perhaps, the Error type should actually be the REDUCTION_ERROR
958  // symbol that we generate when we want to return an error during the
959  // reduction process (rather than here). That way we actually have the
960  // symbol whether or not a we have an error handler for context.
961  Symbol se;
962  switch(context->state) {
963  case tooFewParamsReductionErr: se=TOO_FEW_PARAMS_ERR;break;
964  case tooManyParamsReductionErr: se=TOO_MANY_PARAMS_ERR;break;
965  case signatureMismatchReductionErr: se=SIGNATURE_MISMATCH_ERR;break;
966  case notProcessReductionError: se=NOT_A_PROCESS_ERR;break;
967  case notInSignalContextReductionError: se=NOT_IN_SIGNAL_CONTEXT_ERR;
968  case divideByZeroReductionErr: se=ZERO_DIVIDE_ERR;break;
969  case incompatibleTypeReductionErr: se=INCOMPATIBLE_TYPE_ERR;break;
970  case deadStreamReadReductionErr: se=DEAD_STREAM_READ_ERR;break;
971  case unixErrnoReductionErr:
973  extra = _t_new_str(0,TEST_STR_SYMBOL,strerror(errno));
974  break;
975  case raiseReductionErr:
976  se = *(Symbol *)_t_surface(_t_child(context->node_pointer,1));
977  break;
978  default: raise_error("unknown reduction error: %d",context->state);
979  }
980  T *err = __t_new(ps,se,0,0,1);
981  int *path = _t_get_path(context->node_pointer);
982  _t_new(err,ERROR_LOCATION,path,sizeof(int)*(_t_path_depth(path)+1));
983  free(path);
984  if (extra) {
985  _t_add(err,extra);
986  }
987  // switch the node_pointer to the top of the error handling routine
988  context->node_pointer = _t_child(context->run_tree,3);
989  context->idx = 3;
990  context->parent = context->run_tree;
991 
992  context->state = Eval;
993  }
994  }
995  return context->state;
996 }

1262 
1263  // add to the blocked list
1264  __p_enqueue(q->blocked,qe);
1265  q->contexts_count--;
1266  }

1268  pthread_mutex_unlock(&q->mutex);
1269  debug(D_LOCK,"reduce UNLOCK\n");
1270  };

1275  // expected to inspect the errors of the reduced processes.
1276  debug(D_REDUCE+D_REDUCEV,"Ending reduce\n");

1278 }
